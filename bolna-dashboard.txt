<script>
let contacts = [];
let callQueue = [];
let activeCallIds = {};  // phone -> execution_id
let callStatuses = {};   // phone -> status
let batchIndex = 0;
let pollingInterval = null;
let dispatchInterval = null;
let isCalling = false;

const BATCH_SIZE = 10;
const BATCH_DELAY_MS = 10 * 60 * 1000; // 10 minutes
const POLL_INTERVAL_MS = 10000; // poll every 10s

// ── File upload ──────────────────────────────────────────────
const dropZone = document.getElementById('dropZone');
const fileInput = document.getElementById('fileInput');

dropZone.addEventListener('click', () => fileInput.click());
dropZone.addEventListener('dragover', e => { e.preventDefault(); dropZone.classList.add('drag-over'); });
dropZone.addEventListener('dragleave', () => dropZone.classList.remove('drag-over'));
dropZone.addEventListener('drop', e => {
  e.preventDefault();
  dropZone.classList.remove('drag-over');
  const file = e.dataTransfer.files[0];
  if (file) handleFile(file);
});
fileInput.addEventListener('change', e => { if (e.target.files[0]) handleFile(e.target.files[0]); });

function handleFile(file) {
  const reader = new FileReader();
  reader.onload = e => {
    try {
      const data = new Uint8Array(e.target.result);
      const wb = XLSX.read(data, { type: 'array' });
      const ws = wb.Sheets[wb.SheetNames[0]];
      const rows = XLSX.utils.sheet_to_json(ws, { defval: '' });
      parseContacts(rows);
    } catch(err) {
      alert('Could not parse file: ' + err.message);
    }
  };
  reader.readAsArrayBuffer(file);
}

function parseContacts(rows) {
  const nameKeys = ['name', 'Name', 'NAME', 'Full Name', 'full_name'];
  const phoneKeys = ['phone', 'Phone', 'phone number', 'Phone Number', 'PHONE', 'mobile', 'Mobile', 'contact', 'Contact', 'phone_number'];

  let nameKey = null, phoneKey = null;
  if (rows.length > 0) {
    const keys = Object.keys(rows[0]);
    nameKey = keys.find(k => nameKeys.some(n => k.toLowerCase().includes(n.toLowerCase().trim())));
    phoneKey = keys.find(k => phoneKeys.some(n => k.toLowerCase().includes(n.toLowerCase().trim())));
  }

  if (!nameKey || !phoneKey) {
    alert(`Could not find Name/Phone columns.\nFound columns: ${Object.keys(rows[0] || {}).join(', ')}\n\nPlease ensure your sheet has columns named "Name" and "Phone Number" (or similar).`);
    return;
  }

  contacts = rows
    .map((r, i) => {
      let phone = String(r[phoneKey] || '').trim();
      // Auto-prefix logic for +91
      let cleaned = phone.replace(/[^\d+]/g, ''); 
      if (!cleaned.startsWith('+')) {
        if (cleaned.length === 10) {
          phone = '+91' + cleaned;
        } else if (cleaned.length === 12 && cleaned.startsWith('91')) {
          phone = '+' + cleaned;
        } else if (cleaned.length > 5) {
          // For any other digit-only number, default to +91
          phone = '+91' + cleaned;
        }
      } else {
        phone = cleaned;
      }

      return {
        id: i,
        name: String(r[nameKey] || '').trim(),
        phone: phone,
        status: 'pending',
        executionId: null
      };
    })
    .filter(c => c.name && c.phone);

  if (contacts.length === 0) {
    alert('No valid contacts found in the file.');
    return;
  }

  callStatuses = {};
  contacts.forEach(c => callStatuses[c.id] = 'pending');
  renderTable();
  document.getElementById('tableSection').style.display = 'block';
  document.getElementById('actionBar').style.display = 'flex';
  document.getElementById('countBadge').textContent = `${contacts.length} contacts`;
}

function renderTable() {
  const tbody = document.getElementById('contactsBody');
  tbody.innerHTML = contacts.map((c, i) => `
    <tr id="row-${c.id}">
      <td class="td-phone">${i + 1}</td>
      <td class="td-name">${escHtml(c.name)}</td>
      <td class="td-phone">${escHtml(c.phone)}</td>
      <td><span class="status-pill status-${c.status}" id="status-${c.id}">${statusLabel(c.status)}</span></td>
    </tr>
  `).join('');
}

function updateRowStatus(id, status) {
  contacts[id].status = status;
  const el = document.getElementById(`status-${id}`);
  if (el) {
    el.className = `status-pill status-${status}`;
    el.textContent = statusLabel(status);
  }
  updateProgressUI();
}

function statusLabel(s) {
  return { pending: 'Pending', queued: 'Queued', calling: 'Calling…', completed: 'Completed', called: 'Called', failed: 'Failed' }[s] || s;
}

function escHtml(s) {
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}

// ── Calling Logic ──────────────────────────────────────────────
async function startCalling() {
  const apiKey = document.getElementById('apiKey').value.trim();
  const agentId = document.getElementById('agentId').value.trim();
  if (!apiKey || !agentId) { alert('Please enter your Bolna API Key and Agent ID first.'); return; }
  if (isCalling) return;
  if (contacts.length === 0) { alert('No contacts loaded.'); return; }

  isCalling = true;
  const btn = document.getElementById('makeCallBtn');
  btn.disabled = true;
  btn.innerHTML = `<div class="pulse"></div> Calling in Progress…`;

  callQueue = contacts.filter(c => c.status === 'pending' || c.status === 'failed').map(c => c.id);
  batchIndex = 0;

  document.getElementById('progressSection').style.display = 'block';
  document.getElementById('completedBanner').style.display = 'none';
  document.getElementById('statTotal').textContent = contacts.length;
  updateProgressUI();

  addLog('Calls started — batching 10 every 10 minutes', 'info');
  await dispatchNextBatch(apiKey, agentId);
}

async function dispatchNextBatch(apiKey, agentId) {
  if (callQueue.length === 0) {
    addLog('All calls dispatched, monitoring statuses…', 'info');
    startPolling(apiKey);
    return;
  }

  const batch = callQueue.splice(0, BATCH_SIZE);
  addLog(`Dispatching batch of ${batch.length} call(s)…`, 'info');

  for (const id of batch) {
    const contact = contacts[id];
    updateRowStatus(id, 'queued');
    try {
      const execId = await makeCall(apiKey, agentId, contact.phone);
      contact.executionId = execId;
      updateRowStatus(id, 'calling');
      addLog(`✓ Call queued: ${contact.name} (${contact.phone}) → exec ${execId.slice(0,8)}…`, 'ok');
    } catch(err) {
      updateRowStatus(id, 'failed');
      addLog(`✗ Failed to call ${contact.name}: ${err.message}`, 'err');
    }
    await sleep(600);
  }

  startPolling(apiKey);

  if (callQueue.length > 0) {
    addLog(`Next batch in 10 minutes (${callQueue.length} remaining)…`, 'info');
    setTimeout(() => dispatchNextBatch(apiKey, agentId), BATCH_DELAY_MS);
  }
}

async function makeCall(apiKey, agentId, phone) {
  const res = await fetch('https://api.bolna.ai/call', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${apiKey}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      agent_id: agentId,
      recipient_phone_number: phone
    })
  });

  if (!res.ok) {
    const txt = await res.text();
    throw new Error(`HTTP ${res.status}: ${txt.slice(0,100)}`);
  }

  const data = await res.json();
  if (!data.execution_id) throw new Error('No execution_id returned');
  return data.execution_id;
}

function startPolling(apiKey) {
  if (pollingInterval) return; // already polling
  pollingInterval = setInterval(() => pollStatuses(apiKey), POLL_INTERVAL_MS);
  pollStatuses(apiKey); // immediate first poll
}

async function pollStatuses(apiKey) {
  const inProgress = contacts.filter(c => c.executionId && (c.status === 'calling' || c.status === 'queued'));
  if (inProgress.length === 0) {
    clearInterval(pollingInterval);
    pollingInterval = null;
    checkIfAllDone();
    return;
  }

  for (const contact of inProgress) {
    try {
      const res = await fetch(`https://api.bolna.ai/executions/${contact.executionId}`, {
        headers: { 'Authorization': `Bearer ${apiKey}` }
      });
      if (!res.ok) continue;
      const data = await res.json();
      const s = data.status;

      const statusLower = s?.toLowerCase() || '';
      const isSuccess = ['completed', 'no answer', 'no_answer', 'call disconnected', 'call_disconnected', 'busy'].includes(statusLower);

      if (isSuccess) {
        updateRowStatus(contact.id, 'called');
        addLog(`✓ Called: ${contact.name}`, 'ok');
      } else if (['failed', 'error', 'cancelled'].includes(statusLower)) {
        updateRowStatus(contact.id, 'failed');
        addLog(`✗ Failed: ${contact.name} — ${data.error_message || s}`, 'err');
      }
      // still queued/in_progress → keep polling
    } catch(e) { /* ignore transient network errors */ }
    await sleep(300);
  }

  checkIfAllDone();
}

function checkIfAllDone() {
  const done = contacts.filter(c => c.status === 'completed' || c.status === 'called').length;
  const failed = contacts.filter(c => c.status === 'failed').length;
  const pending = contacts.filter(c => c.status === 'pending' || c.status === 'calling' || c.status === 'queued').length;

  if (pending === 0 && callQueue.length === 0) {
    clearInterval(pollingInterval);
    pollingInterval = null;
    showCompleted(done, failed);
  }
}

function showCompleted(done, failed) {
  document.getElementById('completedBanner').style.display = 'block';
  document.getElementById('completedSummary').textContent =
    `${done} call(s) completed successfully${failed > 0 ? `, ${failed} failed` : ''}.`;
  const btn = document.getElementById('makeCallBtn');
  btn.disabled = false;
  btn.innerHTML = `<span>📞</span> Make Calls`;
  isCalling = false;
  addLog(`All done. ${done} called, ${failed} failed.`, 'ok');
}

function updateProgressUI() {
  const total = contacts.length;
  const done = contacts.filter(c => c.status === 'completed' || c.status === 'called').length;
  const active = contacts.filter(c => c.status === 'calling' || c.status === 'queued').length;
  const failed = contacts.filter(c => c.status === 'failed').length;

  document.getElementById('statTotal').textContent = total;
  document.getElementById('statDone').textContent = done;
  document.getElementById('statActive').textContent = active;
  document.getElementById('statFailed').textContent = failed;

  const pct = total > 0 ? Math.round(((done + failed) / total) * 100) : 0;
  document.getElementById('progressFill').style.width = pct + '%';
}

function addLog(msg, type = 'info') {
  const box = document.getElementById('logBox');
  const ts = new Date().toLocaleTimeString();
  const el = document.createElement('div');
  el.className = `log-line log-${type}`;
  el.innerHTML = `<span class="ts">${ts}</span><span class="msg">${escHtml(msg)}</span>`;
  box.appendChild(el);
  box.scrollTop = box.scrollHeight;
}

function sleep(ms) { return new Promise(r => setTimeout(r, ms)); }
